!ZS_SCRIPT(1.0)

class Elemental
{
	String Name = "NONE"
	Set<key Elemental, float> [] DamageMultipliers = 0


	// Must be there for the Array.Get to be usable on Set group
	bool operator== (Elemental a)
	{
		return this.Name == a.Name
	}
}

{
	Name = "Fire"
	DamageMultipliers =

		<Ice, 3.0>
		<Water, 0.4>
		<Thunder, 1.0>
		<Earth, 0.7>
		<Metal, 0.2>
	}
}
